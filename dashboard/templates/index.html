const socket = io(); socket.on('update_data', function (data) { // Actualizar Balance y Bias if (data.balance) document.getElementById('balance').innerText = `${data.balance} USDT`; if (data.bias) { const el = document.getElementById('bias-trend'); el.innerText = data.bias; el.className = 'card-value ' + (data.bias === 'LONG' ? 'positive' : 'negative'); } // Actualizar Estadísticas D/W/M if (data.stats) { updateStat('daily', data.stats.daily); updateStat('weekly', data.stats.weekly); updateStat('monthly', data.stats.monthly); } // Actualizar Posiciones y Tablas if (data.positions) updatePositionsTable(data.positions); if (data.history) updateHistoryTable(data.history); }); socket.on('new_log', function (data) { const logsDiv = document.getElementById('logs'); const entry = document.createElement('div'); entry.className = 'log-entry ' + (data.type || 'log-info'); entry.innerText = `[${new Date().toLocaleTimeString()}] ${data.message}`; logsDiv.appendChild(entry); logsDiv.scrollTop = logsDiv.scrollHeight; }); function updateStat(period, stats) { const pnlEl = document.getElementById(`${period}-pnl`); const pnlUsdtEl = document.getElementById(`${period}-pnl-usdt`); pnlEl.innerText = `${stats.pnl_pct.toFixed(2)}%`; pnlUsdtEl.innerText = `${stats.total_pnl >= 0 ? '+' : ''}${stats.total_pnl.toFixed(2)} USDT`; const colorClass = stats.pnl_pct >= 0 ? 'positive' : 'negative'; pnlEl.className = 'card-value ' + colorClass; pnlUsdtEl.className = colorClass; document.getElementById(`${period}-wins`).innerText = stats.wins; document.getElementById(`${period}-losses`).innerText = stats.losses; } function updatePositionsTable(positions) { const body = document.getElementById('positions-body'); body.innerHTML = ''; if (positions.length === 0) { body.innerHTML = 'Sin posiciones abiertas'; return; } positions.forEach(p => { const pnl_pct = parseFloat(p.unrealisedPnlPct || 0) * 100; const row = ` ${p.symbol} ${p.side === 'Buy' ? 'LONG' : 'SHORT'} ${p.leverage}x ${pnl_pct.toFixed(2)}% ${parseFloat(p.unrealisedPnl).toFixed(2)} USDT `; body.innerHTML += row; }); } function updateHistoryTable(history) { const body = document.getElementById('history-body'); body.innerHTML = ''; history.forEach(t => { const row = ` ${t.close_time} ${t.symbol} ${t.side} ${t.entry_price} ${t.exit_price} ${t.pnl_pct.toFixed(2)}% ${t.pnl_pct >= 0 ? '+' : ''}${t.pnl_usdt.toFixed(2)} USDT ${t.reason} `; body.innerHTML += row; }); } // Control Buttons function startBot() { socket.emit('control_bot', { action: 'start' }); } function stopBot() { socket.emit('control_bot', { action: 'stop' }); } function resetStats() { if(confirm('¿Seguro que quieres resetear los contadores de la sesión?')) socket.emit('control_bot', { action: 'reset' }); }
